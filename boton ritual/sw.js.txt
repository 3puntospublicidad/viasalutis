self.addEventListener('install', e => {
  e.waitUntil(
    caches.open('via-salutis').then(cache => {
      return cache.addAll([
        'index.html',
        'styles/styles.css',
        'assets/escudo-via-salutis.png',
        // Agrega aquÃ­ los demÃ¡s archivos
      ]);
    })
  );
});

self.addEventListener('fetch', e => {
  e.respondWith(
    caches.match(e.request).then(response => {
      return response || fetch(e.request);
    })
  );
});
